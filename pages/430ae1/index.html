# 六、固定导航栏

<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>固定导航栏</title>
  </head>
  <style>
    * {
      margin: 0;
      padding: 0;
    }
    nav {
      position: fixed;
      top: 0;
      left: 0;
      background-color: #2496ed;
      width: 100%;
      height: 60px;
      line-height: 60px;
      color: #fff;
      text-align: center;
    }
    div {
      margin-top: 65px;
      height: 2000px;
      background-color: #a3a8b1;
      text-align: center;
    }
  </style>
  <body>
    <nav>顶部导航栏</nav>
    <div>内容区域</div>
  </body>
</html>

# 七、动态固定导航栏

<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>动态固定导航栏</title>
  </head>
  <style>
    * {
      margin: 0;
      padding: 0;
    }
    nav.top {
      position: fixed;
      top: 0;
      left: 0;
      background-color: #2496ed;
      width: 100%;
      height: 60px;
      line-height: 60px;
      color: #fff;
      text-align: center;
    }
    nav.sticky {
      position: sticky;
      height: 100px;
      top: 60px;
      background-color: #0d515a;
      color: #fff;
      text-align: center;
    }
    div {
      height: 200px;
      border: 1px solid #37383b;
      background-color: #a3a8b1;
      text-align: center;
    }
    div.first {
      margin-top: 65px;
    }
  </style>
  <body>
    <nav class="top">顶部导航栏</nav>
    <div class="first">内容区域1</div>
    <nav class="sticky">动态导航栏</nav>
    <div>内容区域2</div>
    <div>内容区域3</div>
    <div>内容区域4</div>
    <div>内容区域5</div>
    <div>内容区域6</div>
    <div>内容区域7</div>
  </body>
</html>
