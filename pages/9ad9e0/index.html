JS之this的绑定

一次性吃透 JS 中 this 的绑定吧

# 一、使用 this 的原因

不使用 this：

  1. 需要显示的定义传参，参数过多会难以管理
function getName(student) {
  console.log(student.name);
}
function getAge(student) {
  console.log(student.age);
}
var studentA = {
  name: 'alex',
  age: '20'
};
var studentB = {
  name: 'leo',
  age: '21'
};
getName(studentA);
getName(studentB);
getAge(studentA);
getAge(studentB);

2.this 并不指向函数的词法作用域

在 JavaScript 内部，作用域确实和对象类似，可见的标识符都是它的属性。但是作用域“对象”无法通过 JavaScript

代码访问，它存在于 JavaScript 引擎内部。

# 二、this 的绑定

this 是在函数被调用时才发生的绑定，它指向什么完全取决于函数在哪里被调用

# 2.1 默认绑定

var a = 2;
function foo() {
  return this.a;
}
foo(); // 2 相当于 this.foo();

// 严格模式下无法绑定 会报错
function foo() {
  'use strict';
  console.log(this.a);
}
var a = 2;
foo(); // TypeError: this is undefined

# 2.2 隐式绑定

function foo() {
  return this.a;
}
var obj = {
  a: 1,
  foo: foo // 若将obj.foo赋值给其他变量再调用，会导致“隐式丢失”
};
obj.foo(); // 1

// 隐式丢失 1
function foo() {
  console.log(this.a);
}
var obj = {
  a: 2,
  foo: foo
};
var bar = obj.foo; // 函数别名！
var a = 'oops, global'; // a 是全局对象的属性
bar(); // "oops, global"

// 隐式丢失 2
function foo() {
  console.log(this.a);
}
function doFoo(fn) {
  // fn 其实引用的是foo
  fn(); // <-- 调用位置！ 相当于 this.fn()  this指向window
}
var obj = {
  a: 2,
  foo: foo
};
var a = 'oops, global'; // a 是全局对象的属性
doFoo(obj.foo); // "oops, global"

# 2.3 显示绑定

如果你传入了一个原始值（字符串类型、布尔类型或者数字类型）来当作 this 的绑定对象，这个原始值会被转换成它的对象形式（也就是 new String(..)、new Boolean(..) 或者 new Number(..)）。这通常被称为“装箱”。

// 1. call(context,a,b,c);
// 2. apply(context,[a,b,c]);
// 3. let fn = bind(context); fn(a,b,c);
function foo() {
  return this.a;
}
var obj = {
  a: 1
};
foo.call(obj); // 1

// 硬绑定---解决隐式丢失
function foo() {
  console.log(this.a);
}
var obj = {
  a: 1
};
var a = 2;
var bar = function() {
  foo.call(obj);
};
bar(); // 1
setTimeout(bar, 100); // 1
bar.call(window); // 1 硬绑定的bar 不可能再修改它的this

bind 的实现

Function.prototype.bind = function(context) {
  if (typeof this !== 'function') {
    throw new TypeError('Function.prototype.bind - what is trying to be bound is not callable');
  }

  var self = this;
  var args = Array.prototype.slice.call(arguments, 1);

  var tempFn = function() {};
  var resultFn = function() {
    var innerArgs = Array.prototype.slice.call(arguments);
    self.apply(this instanceof tempFn ? this : context, args.concat(innerArgs));
  };
  tempFn.prototype = this.prototype;
  resultFn.prototype = new tempFn();
  return resultFn;
};

call 和 apply 的区别、性能对比

  1. Function.prototype.apply 和 Function.prototype.call 的作用是一样的，区别在于传入参数的不同：

    第一个参数：都是指定函数体内 this 的指向；

    第二个参数：apply 是传入带下标的集合，数组或者类数组，apply 把它传给函数作为参数，call 从第二个开始传入的参数是不固定的，都会传给函数作为参数。

  2. call 比 apply 的性能要好，平常可以多用 call, call 传入参数的格式正是内部所需要的格式

var bar = new foo(2);

new 的实现原理

function myNew(fn,...args){  // 1. 创建新对象obj  var obj = {};     // 2. 若fn.prototype为对象类型，则obj的[[Prototype]]属性指向fn.prototype，否则指向Object.prototype  // 前两步骤同： var obj = Object.create(fn.prototype);  obj.prototype = fn.prototype;    // 3. 调用fn函数，并将this指向新创建的obj对象  var result = fn.apply(obj,args);  // 4. 若fn函数return一个对象，则返回该对象，否则返回新创建的obj对象  return result instanceof Object ? result : obj;}

# 2.5 箭头函数

根据外层（函数或者全局）作用域来决定 this，箭头函数的绑定无法被修改

function foo() {  // 返回一个箭头函数  return (a) => {    //this 继承自foo()    console.log( this.a );  };}var obj1 = {  a:2};var obj2 = {  a:3};var bar = foo.call( obj1 ); // 箭头函数的this被绑定到obj1，之后将不会再改变bar.call( obj2 ); // 2, 不是3 ！
1

# 三、总结

3.1 函数是否在 new 中调用（new 绑定）？如果是的话 this 绑定的是新创建的对象。

var bar = new foo();
1

3.2 函数是否通过 call、apply（显式绑定）或者硬绑定调用？如果是的话，this 绑定的是指定的对象。

var bar = foo.call(obj2);
1

3.3 函数是否在某个上下文对象中调用（隐式绑定）？如果是的话，this 绑定的是那个上下文对象。

var bar = obj1.foo();
1

3.4 如果都不是的话，使用默认绑定。如果在严格模式下，就绑定到 undefined，否则绑定到全局对象。

var bar = foo();
