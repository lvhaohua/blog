JS之防抖（debounce）与节流（throttle）

防抖节流是常用的优化手段，例如滚动，请求等场景

# 一、防抖（debounce）

最后一次触发高频事件的单位时间（delay）后执行回调函数（fn），如果在单位时间（delay）内又被触发，则重新计时。

应用场景：

  1. search 搜索联想，用户在不断输入值时，用防抖来节约请求资源
  2. window 触发 resize 的时候，不断的调整浏览器窗口大小会不断的触发这个事件，用防抖来让其只触发一次
function debounce(fn, delay) {
  let timeTag = null; // 创建一个标记用来存放定时器
  return function() {
    clearTimeout(timeTag); // 每当用户输入的时候把前一个定时器 clear 掉
    timeTag = setTimeout(() => {
      // 创建新的计时器，重新计时
      fn.applay(this, arguments);
    }, delay);
  };
}

function fetch(content) {
  console.log('fetch...' + content);
}

let ipt = document.getElementById('ipt');
ipt.addEventListener('input', (e) => {
  debounce(fetch, 500)(e.target.value);
});

# 二、节流（throttle）

每隔单位时间（delay）后执行一次回调函数（fn）。

应用场景：

  1. 鼠标不断点击触发，mousedown(单位时间内只触发一次)
  2. 监听滚动事件，比如是否滑到底部自动加载更多，用 throttle 来判断
function throttle(fn, delay) {
  let canRun = true;
  return function() {
    if (!canRun) return;

    canRun = false;
    setTimeout(() => {
      fn.apply(this, arguments);
      canRun = true;
    }, delay);
  };
}

function fetch(content) {
  console.log('ajax request ' + content);
}

let ipt = document.getElementById('ipt');
ipt.addEventListener('keyup', function(e) {
  throttle(fetch, 1000)(e.target.value);
});
