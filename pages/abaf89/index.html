JS之深拷贝与浅拷贝

深拷贝与浅拷贝也是开发中常见知识点，面试也会常问哦

# 一、浅拷贝 shallow copy

不同的引用指向同一片内存

# 1.1 Object.assign

第一层深拷贝

第二层开始浅拷贝

let obj = {
  name: 'alex',
  age: 30,
  friends: {
    name: 'bob',
    age: 30
  }
};
let obj_copy = Object.assign({}, obj);
obj_copy.name = 'candy'; // { name: "candy", age: 30, friends: { name: "bob", age: 30 } }
console.log(obj); // { name: "alex", age: 30, friends: { name: "bob", age: 30 } }

obj_copy.friends.age = 32;
console.log(obj); // { name: "alex", age: 30, friends: { name: "bob", age: 32 } }

# 2.2 递归

function isObject(target) {
  return typeof target === 'object' && target != null;
}

function deepClone(target) {
  if (!isObject(target)) return target;
  let cloneObj = {};
  Object.keys(target).forEach((key) => {
    if (isObject(target[key])) {
      cloneObj[key] = deepClone(target[key]);
    } else {
      cloneObj[key] = target[key];
    }
  });
  return cloneObj;
}

  1. 循环引用
  2. 重复引用
  3. Symbol

// 1. 解决循环引用问题
// 2. 解决重复引用问题
function isObject(target) {
  return typeof target === 'object' && target != null;
}

function deepClone(target) {
  let store = new WeakMap();
  function clone(target) {
    if (!isObject(target)) return target;
    if (store.has(target)) return store.get(target);

    let target_copy = Array.isArray(target) ? [] : {};
    store.set(target, target_copy);

    Object.keys(target).forEach((key) => {
      if (isObject(target[key])) {
        target_copy[key] = clone(target[key]);
      } else {
        target_copy[key] = target[key];
      }
    });
    return target_copy;
  }
  return clone(target);
}

// 4. 解决递归爆栈问题
function cloneDeep(target) {
  const root = {};

  // 栈
  const loopList = [
    {
      parent: root,
      key: undefined,
      data: target
    }
  ];

  while (loopList.length) {
    // 广度优先
    const node = loopList.pop();
    const parent = node.parent;
    const key = node.key;
    const data = node.data;

    // 初始化赋值目标，key为undefined则拷贝到父元素，否则拷贝到子元素
    let res = parent;
    if (typeof key !== 'undefined') {
      res = parent[key] = {};
    }

    for (let k in data) {
      if (data.hasOwnProperty(k)) {
        if (typeof data[k] === 'object') {
          // 下一次循环
          loopList.push({
            parent: res,
            key: k,
            data: data[k]
          });
        } else {
          res[k] = data[k];
        }
      }
    }
  }

  return root;
}
