# 7. entries keys values Symbol.iterator （均为 ES6）

// 1. entries 返回一个数组迭代器对象，该对象包含数组中每个索引的键值对 {value:[item_key,item_value],done:isLast}
var array = ['a', 'b', 'c'];
var iterator = array.entries();
console.log(iterator.next().value); // [0, "a"]
console.log(iterator.next().value); // [1, "b"]
console.log(iterator.next().value); // [2, "c"]
console.log(iterator.next().value); // undefined, 迭代器处于数组末尾时, 再迭代就会返回undefined

// 2. keys 返回一个数组索引的迭代器 {value:item_key,done:isLast}
var array = ['abc', 'xyz'];
var iterator = array.keys();
console.log(iterator.next()); // Object {value: 0, done: false}
console.log(iterator.next()); // Object {value: 1, done: false}
console.log(iterator.next()); // Object {value: undefined, done: false}

// 3. values 返回一个数组值的迭代器 {value:item_value,done:isLast}
var array = ['abc', 'xyz'];
var iterator = array.values();
console.log(iterator.next().value); //abc
console.log(iterator.next().value); //xyz

// 4. Symbol.iterator 同values
var array = ['abc', 'xyz'];
var iterator = array[Symbol.iterator]();
console.log(iterator.next().value); // abc
console.log(iterator.next().value); // xyz

function unique(arr) {
  var arr2 = [];

function unique(arr) {
  var arr2 = [];

// 双层循环嵌套 查找后面是否有重复的 无法去掉NaN和{}
function unique(arr) {
  for (var i = 0; i < arr.length; i++) {
    for (var j = i + 1; j < arr.length; j++) {
      if (arr[i] === arr[j]) {
        arr.splice(j, 1);
      }
    }
  }
  return arr;
}
// sort sort后通过检测后面一个是否等于前面一个来排除 无法去掉NaN和{} 而且会乱序
function unique(arr) {
  var arr2 = arr.sort();
  var arr3 = [arr2[0]];
  for (var i = 1; i < arr2.length; i++) {
    if (arr2[i] !== arr3[i - 1]) {
      arr3.push(arr2[i]);
    }
  }
  return arr3;
}
// hasOwnProperty  可以全部去重成功
function unique(arr) {
  var temp = {};
  var resultArr = arr.filter(function(item) {
    if (temp.hasOwnProperty(typeof item + item)) {
      return false;
    } else {
      temp[typeof item + item] = true;
      return true;
    }
  });
  return resultArr;
}

# 7. map 怎么实现的

if (!Array.prototype.map) {
  Array.prototype.map = function(callback, thisArg) {
    var oldArray = Object(this), // 拿到旧的数组
      context,
      resultArray = [], // 声明返回的新的数组
      i = 0; // 循环用的索引
    if (typeof callback !== 'function') {
      throw new TypeError('xxx');
    }

    if (arguments.length > 1) {
      context = arguments[1]; // Array.prototype.slice.call(arguments,1) 获取thisArg
    }

    while (i < oldArray.length) {
      if (i in oldArray) {
        var item = oldArray[i];
        resultArray[i] = callback.call(context, item, i, oldArray);
      }
      i++;
    }

    return resultArray;
  };
}
