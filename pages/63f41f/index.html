JS之继承

介绍了 JS 种七种继承方式

# 一、原型链继承

弊端：

  1. 父类的实例属性会被多个子类的实例共享
  2. 无法在不影响所有对象实例的情况下，给父类的构造函数传参
function Parent() {
  this.parentName = 'parentName';
  this.favFruit = ['apple', 'banana', 'orange']; // 会被子类实例共享
}
Parent.prototype.getParentName = function() {
  return this.parentName;
};

function Child() {
  this.childName = 'childName';
}
Child.prototype = new Parent();
Child.prototype.getChildName = function() {
  return this.childName;
};

var childA = new Child();
var childB = new Child();
childA.getParentName(); // "parentName"
childA.favFruit.push('watermolen');
childA.favFruit; // ["apple", "banana", "orange", "watermolen"]
childB.favFruit; // ["apple", "banana", "orange", "watermolen"]

# 二、借用构造函数

弊端：

  1. 在构造函数中定义的方法（getFavFruit）无法复用，每次 new 新的实例都会重新构建一次方法
function Parent(name) {
  this.name = name;
  this.colors = ['green', 'blue'];
  this.getFavFruits = function() {
    // 该方法无法复用，每次new新的实例都会重新构建一次方法
    return ['apple'];
  };
}

function Child(name, age) {
  Parent.call(this, name); // 可以向父类传参
  this.age = age;
}

var childA = new Child('childA', 20);
var childB = new Child('childB', 21);
childA.colors.push('yellow');
childA.colors; // ["green", "blue", "yellow"] 解决了“原型链继承”父类属性被子类共享的问题
childB.colors; // ["green", "blue"]
childA instanceof Parent; // false

# 三、组合继承

注：组合“借用构造继承”和“原型链继承”--最常用的继承方式

弊端：

  1. 调用了两次父类构造函数
function Parent(name) {
  this.name = name;
  this.colors = ['green', 'blue'];
}
Parent.prototype.getName = function() {
  return this.name;
};

function Child(name, age) {
  Parent.call(this, name); // new的时候 第二次调用
  this.age = age;
}
Child.prototype = new Parent(); // 第一次调用
Child.prototype.constructor = Child;
Child.prototype.getAge = function() {
  return this.age;
};

var childA = new Child('ChildA', 20);
var childB = new Child('childB', 22);

childA.colors.push('yellow');
childA.colors; // ["green", "blue", "yellow"]
childB.colors; // ["green", "blue"]

childA.getName(); // "ChildA"
childB.getName(); // "ChildB"

childA instanceof Parent; // true
childA instanceof Child; // true

# 四、原型式继承

弊端：

  1. 子类共享父类的属性
function createObj(obj) {
  function F() {}
  F.prototype = obj;
  return new F();
}

var parent = {
  name: 'parentName',
  colors: ['green', 'blue']
};

var childA = createObj(parent);
childA.name = 'childAName';
childA.colors.push('yellow'); // ["green", "blue", "yellow"]

var childB = createObj(parent);
childB.name = 'childBName';
childB.colors.push('white'); // ["green", "blue", "yellow", "white"]

# 五、寄生式继承

注：在原型式继承基础上封装一个增强函数

弊端：

  1. 每次创建对象都会重复创建一次增强函数（getName）
function createObj(obj) {
  var obj = Object.create(obj);
  obj.getName = function() {
    // 增强对象  每次调用createObj都会重新创建该增强函数
    return this.name;
  };
  return obj;
}

var parent = {
  name: 'parentName',
  colors: ['green', 'blue']
};

var child = createObj(parent);
child.name = 'childName';
child.getName(); // "childName"

# 六、寄生组合式继承

注：最理想的继承模式

// 1.原型继承原型 2.构造属性继承构造属性
function Parent(name) {
  this.name = name;
}
Parent.prototype.getName = function() {
  return this.name;
};

function Child(name, age) {
  // 2.构造器属性继承
  Parent.call(this, name);
  this.age = age;
}

function createObj(child, parent) {
  // 1.原型继承 Child.prototype == Parent.prototype的副本
  var temp = Object.create(parent.prototype);
  temp.prototype.constructor = child;
  child.prototype = temp;
}
createObj(Child, Parent);
// 要增强Child的对象能力只能在createObj之后
Child.prototype.getAge = function() {
  return this.age;
};

var childA = new Child('childAName', 20);

# 七、Class 继承

class Parent {
  constructor(name, age) {
    this.name = name;
    this.age = age;
  }

  info() {
    console.log(`My name is ${this.name}, age is ${this.age}`);
  }
}

class Child extends Parent {
  constructor(name, age, gender) {
    super(name, age);
    this.gender = gender;
  }

  info() {
    console.log(`My name is ${this.name}, age is ${this.age}, gender is ${this.gender}`);
  }
}

const parent = new Parent('parent', 50);
const child = new Child('parent', 20, 'male');
